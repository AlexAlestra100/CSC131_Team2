function findYear(yearStart, yearEnd)
{
    var nstart = parseFloat(yearStart.value);
    var nend = parseFloat(yearEnd.value);

    if(nstart < 1927)
    {
        nstart = 1927;
    }
    else if(nend > 2017)
    {
        nend = 2017;
    }
     
    var years = [];
    
    for(var i = nstart; i <= nend; i++)
    {
        years.push(i);
    }
    return years;
}

function winOlose(bx)
{
    if(bx.value == "Win")
    {
        return true;
    }
    else if(bx.value == "Lose")
    {
        return false;
    }
    else
    {
        return null;
    }
}
